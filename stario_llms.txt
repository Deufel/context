# Stario Reference

Python 3.14+. Real-time hypermedia. No JS frameworks.

```bash
pip install stario
```

---

## Handler

Every handler: `async def handler(c: Context, w: Writer) -> None`

- `c` reads request data
- `w` sends responses

```python
from stario import Stario, RichTracer

async def main():
    with RichTracer() as tracer:
        app = Stario(tracer)
        app.get("/", home)
        await app.serve(port=8000)
```

---

## Context (`c`)

**Request:**
```python
c.req.method          # "GET", "POST"
c.req.path            # "/users/123"
c.req.query.get("q")  # query param
c.req.tail            # path remainder for /* routes
c.req.headers.get("X-Custom")
c.req.cookies.get("session")
```

**Body:**
```python
await c.req.json()    # dict
await c.req.form()    # multipart/urlencoded
await c.req.body()    # bytes
```

**Signals** (Datastar state from client):
```python
signals = await c.signals()           # dict
signals = await c.signals(MyClass)    # dataclass
```

**Tracing:**
```python
c["user_id"] = 42                     # span attribute
with c.step("db.query") as s:         # child span
    s["rows"] = count
c("cache.miss")                       # event
```

**State** (middleware data):
```python
c.state["user"] = user
```

---

## Writer (`w`)

**One-shot:**
```python
w.html(element)           # HTML response
w.json({"ok": True})      # JSON
w.text("plain")           # text
w.redirect("/path")       # 307 redirect
w.empty(204)              # no content (execution continues!)
```

**SSE streaming:**
```python
w.patch(element)                              # morph by element id
w.patch(el, selector="#target")               # morph into selector
w.patch(el, selector="#list", mode="append")  # append
w.patch("", selector="#row", mode="remove")   # remove
w.sync({"count": 5})                          # update signals
w.navigate("/home")                           # client navigation
w.execute("alert('hi')")                      # run JS
w.remove("#error")                            # remove element
```

**Connection lifecycle:**
```python
async for event, data in w.alive(relay.subscribe("*")):
    w.patch(view())
# code here runs on disconnect
```

**Cookies/headers:**
```python
w.cookie("session", token, httponly=True, secure=True)
w.delete_cookie("session")
w.status(201).header("X-Id", "1").json(data)
```

## HTML

```python
from stario.html import Div, H1, P, A, Ul, Li, Input, Button, Form
```

First arg dict = attributes. Rest = children.

```python
Div({"class": "box", "id": "main"}, H1("Title"), P("Body"))
Div({"style": {"color": "red"}}, "styled")      # style dict
Div({"disabled": True})                          # boolean attr
Div({"class": ["a", "b"]})                       # class="a b"
```

Auto-escapes strings. For trusted HTML:
```python
from stario.html import SafeString
Div(SafeString("<svg>...</svg>"))
```

## Datastar

**Attributes (`data.*`):**
```python
data.signals({"count": 0}, ifmissing=True)  # init state
data.bind("username")                        # two-way bind
data.text("$count")                          # reactive text
data.show("$visible")                        # conditional display
data.show("!$loading")                       # negation
data.on("click", "$count++")                 # expression
data.on("click", at.post("/add"))            # server action
data.on("input", at.get("/search"), debounce=0.3)
data.init(at.get("/subscribe", retry="always"))
data.indicator("loading")                    # $loading=true during fetch
data.computed({"total": "$a + $b"})
data.effect("console.log($count)")
```

**Actions (`at.*`):**
```python
at.get("/path")
at.post("/path")
at.post("/path", include=["field"], selector="#out")
at.put("/path")
at.patch("/path")
at.delete("/path")
```

## Routing

```python
app.get("/", home)
app.post("/users", create)
app.get("/users/*", get_user)     # c.req.tail = remainder

# Sub-router
api = Router()
api.get("/items", list_items)
app.mount("/api", api)

# Middleware
app.use(logger)                   # all routes
app.get("/admin", page, auth)     # per-route (right-to-left)
```

## Relay (Pub/Sub)

```python
from stario import Relay

relay: Relay[str] = Relay()

# Publish (sync, fire-and-forget)
relay.publish("topic", payload)

# Subscribe
async for topic, data in w.alive(relay.subscribe("*")):
    w.patch(render())
```

Wildcards: `"chat.*"` matches `chat.1`, `chat.room`. `"*"` matches all.

## CQRS Pattern

Commands write. Queries read.

```python
# Command: POST → 204 → publish
async def add(c: Context, w: Writer):
    signals = await c.signals()
    items.append(signals["new"])
    relay.publish("refresh", None)
    w.empty()

# Query: SSE → patch on events
async def stream(c: Context, w: Writer):
    w.patch(render_list())
    async for _ in w.alive(relay.subscribe("refresh")):
        w.patch(render_list())
```

## Optimistic UI

1. POST returns `w.empty(204)` immediately, publishes event
2. Subscribe handler appends loading row
3. Loading row has `data.init(at.get("/fetch"))` to lazy-load
4. Fetch handler patches real content

```python
def loading_row(id):
    return Tr({"id": f"row-{id}"},
        data.init(at.get(f"/fetch?id={id}")),
        Td("Loading...")
    )
```

## Error Handling

```python
from stario import HttpException

raise HttpException(404, "Not found")

# Global handler
def handle_404(c, w, exc):
    w.html(Div("Not found"), status=404)

app.on_error(HttpException, handle_404)
```

## Testing

```python
from stario import TestRequest, ResponseRecorder, Context

req = TestRequest(method="POST", body={"name": "Alice"})
rec = ResponseRecorder()
await handler(Context(req), rec)

assert rec.status == 200
assert "Alice" in rec.body
# rec.patches, rec.signals for SSE
```

## Auth Pattern

```python
def require_auth(next_handler):
    async def h(c, w):
        sid = c.req.cookies.get("session")
        if not sid:
            return w.redirect("/login")
        c.state["user"] = await get_user(sid)
        await next_handler(c, w)
    return h

async def login(c: Context, w: Writer):
    signals = await c.signals()
    user = await authenticate(signals["user"], signals["pass"])
    if not user:
        return w.sync({"error": "Invalid"})
    w.cookie("session", create_session(user), httponly=True, secure=True)
    w.navigate("/home")
```

## File Upload

Files arrive Base64 in signals:
```python
signals = await c.signals()
for f in signals.get("files", []):
    data = base64.b64decode(f["contents"].split(",")[1])
    Path(f"uploads/{f['name']}").write_bytes(data)
```

## Dependency Injection

Closures. No magic.

```python
def make_handler(db):
    async def h(c, w):
        w.json(await db.find(c.req.tail))
    return h

app.get("/users/*", make_handler(my_db))
```

## Gotchas

1. `w.empty(204)` doesn't stop execution
2. Python 3.14+ required
3. `{...}` is a set, not dict—don't use as placeholder
4. Datastar CDN version matters: `@1.0.0-RC.7`
5. Full-page morph often simpler than targeted patches
6. Check item exists before patching (race conditions)
